<script>
    /* =================================================================
       A. エンターシーケンス & Web Audio API (音楽解析・自動再生)
    ==================================================================== */
    const entryScreen = document.getElementById('entry-screen');
    const mainContent = document.getElementById('main-content');
    const audio = document.getElementById('main-audio');
    const avatarGlow = document.getElementById('avatar-glow');
    const cursorLight = document.getElementById('js-cursor-light');
    
    let audioCtx, analyser, dataArray, bufferLength;
    let isAudioInitialized = false;
    let currentBass = 0; // 低音域の強さ（0〜255）

    // エンタークリックで音楽自動再生とUI表示
    entryScreen.addEventListener('click', () => {
        // Web Audio APIの初期化
        if (!isAudioInitialized) {
            const AudioContext = window.AudioContext || window.webkitAudioContext;
            audioCtx = new AudioContext();
            analyser = audioCtx.createAnalyser();
            const source = audioCtx.createMediaElementSource(audio);
            source.connect(analyser);
            analyser.connect(audioCtx.destination);
            
            analyser.fftSize = 128; // 解析の解像度
            bufferLength = analyser.frequencyBinCount;
            dataArray = new Uint8Array(bufferLength);
            isAudioInitialized = true;
        }
        
        // 再生開始
        if (audioCtx.state === 'suspended') {
            audioCtx.resume();
        }
        audio.play().catch(e => console.log("再生エラー:", e));

        // UIの切り替え
        entryScreen.style.opacity = '0';
        entryScreen.style.transform = 'scale(1.1)';
        setTimeout(() => {
            entryScreen.style.display = 'none';
            mainContent.style.opacity = '1';
            setTimeout(reveal, 100); // 要素フェードイン
        }, 800);
        
        // クリック波紋発生
        ripples.push(new Ripple(window.innerWidth/2, window.innerHeight/2));
    });


    /* =================================================================
       B. CANVAS EFFECTS (オーディオリアクティブ・ビジュアライザー + 粒子)
    ==================================================================== */
    const canvas = document.getElementById('bg-canvas');
    const ctx = canvas.getContext('2d');
    let particles = [];
    let ripples = [];
    const maxParticles = 70;

    function resizeCanvas() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    class Particle {
        constructor() { this.reset(); }
        reset() {
            this.x = Math.random() * canvas.width;
            this.y = Math.random() * canvas.height;
            this.baseSize = Math.random() * 1.5 + 0.5;
            this.speedX = Math.random() * 0.4 - 0.2;
            this.speedY = Math.random() * 0.5 + 0.2;
            this.alpha = Math.random() * 0.5 + 0.2;
            this.fadeSpeed = Math.random() * 0.01 + 0.005;
        }
        update() {
            // 音楽の低音に合わせて速度が微増
            const bassSpeed = (currentBass / 255) * 1.5;
            this.y += this.speedY + (bassSpeed * 0.5);
            this.x += this.speedX;
            
            this.alpha += this.fadeSpeed;
            if(this.alpha > 0.8 || this.alpha < 0.1) this.fadeSpeed *= -1;

            if (this.y > canvas.height || this.x < 0 || this.x > canvas.width) {
                this.reset();
                this.y = 0;
            }
        }
        draw() {
            // 音楽の低音に合わせてサイズが脈打つ
            const currentSize = this.baseSize + (currentBass / 255) * 3;
            
            ctx.save();
            ctx.globalAlpha = this.alpha;
            ctx.fillStyle = '#ffffff';
            ctx.beginPath();
            ctx.arc(this.x, this.y, currentSize, 0, Math.PI * 2);
            ctx.shadowBlur = 15;
            ctx.shadowColor = '#ffffff';
            ctx.fill();
            ctx.restore();
        }
    }

    class Ripple {
        constructor(x, y) {
            this.x = x; this.y = y;
            this.radius = 0; this.maxRadius = 100;
            this.alpha = 1; this.speed = 4;
        }
        update() {
            this.radius += this.speed;
            this.alpha = 1 - (this.radius / this.maxRadius);
        }
        draw() {
            if(this.alpha <= 0) return;
            ctx.save();
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.strokeStyle = `rgba(255, 255, 255, ${this.alpha})`;
            ctx.lineWidth = 1.5;
            ctx.shadowBlur = 20;
            ctx.shadowColor = '#ffffff';
            ctx.stroke();
            ctx.restore();
        }
    }

    for (let i = 0; i < maxParticles; i++) particles.push(new Particle());

    window.addEventListener('click', (e) => {
        // エンタースクリーン非表示後のみ波紋を生成
        if(entryScreen.style.display === 'none') {
            ripples.push(new Ripple(e.clientX, e.clientY));
        }
    });

    function animate() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // 音楽データの取得
        if (isAudioInitialized && !audio.paused) {
            analyser.getByteFrequencyData(dataArray);
            
            // 低音域（ベース）の平均値を算出
            let sum = 0;
            for(let i = 0; i < 5; i++) sum += dataArray[i];
            currentBass = sum / 5;
            
            // アバターの鼓動エフェクト
            const scale = 1 + (currentBass / 255) * 0.3;
            const glowOpacity = 0.3 + (currentBass / 255) * 0.7;
            avatarGlow.style.transform = `scale(${scale})`;
            avatarGlow.style.opacity = glowOpacity;
            
            // カーソルライトの反応
            cursorLight.style.width = (500 + currentBass) + 'px';
            cursorLight.style.height = (500 + currentBass) + 'px';

            // ビジュアライザー描画 (画面下部)
            ctx.save();
            let barWidth = (canvas.width / bufferLength) * 2;
            let barHeight;
            let x = 0;
            for(let i = 0; i < bufferLength; i++) {
                barHeight = (dataArray[i] / 255) * 150; // 最大高さ150px
                ctx.fillStyle = `rgba(255, 255, 255, ${dataArray[i] / 255 * 0.5})`;
                ctx.fillRect(x, canvas.height - barHeight, barWidth - 1, barHeight);
                x += barWidth;
            }
            ctx.restore();
        } else {
            currentBass = 0;
            avatarGlow.style.transform = `scale(1)`;
        }

        particles.forEach(p => { p.update(); p.draw(); });
        
        ripples = ripples.filter(r => r.alpha > 0);
        ripples.forEach(r => { r.update(); r.draw(); });

        requestAnimationFrame(animate);
    }
    animate();
